<script>
  import { onMount } from 'svelte';

  // semplice directive per animazione entranceUp
  export function fadeUp(node, delay = 0) {
    node.style.opacity = '0';
    node.style.transform = 'translateY(16px)';
    node.style.transition =
      `opacity 0.6s cubic-bezier(0.25,0.1,0.25,1) ${delay}s,
       transform 0.6s cubic-bezier(0.25,0.1,0.25,1) ${delay}s`;

    const observer = new IntersectionObserver(([entry]) => {
      if (entry.isIntersecting) {
        node.style.opacity = '1';
        node.style.transform = 'translateY(0)';
        observer.disconnect();
      }
    }, { threshold: 0.2 });

    observer.observe(node);
    return {
      destroy() { observer.disconnect(); }
    };
  }
</script>

<!-- VideoSlider.svelte -->
<script>
  import { onMount } from 'svelte';

  let isDown = false;
  let startX;
  let scrollLeft;
  let activeVideo = null;
  let wrapper;

  const videos = [
    { id: 1, src: '/media/case1.mp4', label: 'Spot PMI 1' },
    { id: 2, src: '/media/case2.mp4', label: 'Spot PMI 2' },
    // ...
  ];

  function handleMouseDown(e) {
    isDown = true;
    wrapper.classList.add('is-dragging');
    startX = e.pageX - wrapper.offsetLeft;
    scrollLeft = wrapper.scrollLeft;
  }
  function handleMouseLeave() {
    isDown = false;
    wrapper.classList.remove('is-dragging');
  }
  function handleMouseUp() {
    isDown = false;
    wrapper.classList.remove('is-dragging');
  }
  function handleMouseMove(e) {
    if (!isDown) return;
    e.preventDefault();
    const x = e.pageX - wrapper.offsetLeft;
    const walk = (x - startX) * 1.2;
    wrapper.scrollLeft = scrollLeft - walk;
  }

  function openLightbox(video) {
    activeVideo = video;
  }
  function closeLightbox() {
    activeVideo = null;
  }
</script>

<section class="section" id="portfolio">
  <div class="section-inner">
    <h2 class="sec-title">Progetti per PMI come la tua</h2>
    <div
      class="video-slider"
      bind:this={wrapper}
      on:mousedown={handleMouseDown}
      on:mouseleave={handleMouseLeave}
      on:mouseup={handleMouseUp}
      on:mousemove={handleMouseMove}
    >
      {#each videos as v}
        <button class="slider-video" on:click={() => openLightbox(v)}>
          <video muted playsinline>
            <source src={v.src} type="video/mp4" />
          </video>
          <span class="video-label">{v.label}</span>
        </button>
      {/each}
    </div>
  </div>

  {#if activeVideo}
    <div class="lightbox" on:click|self={closeLightbox}>
      <div class="lightbox-inner">
        <button class="lightbox-close" on:click={closeLightbox}>×</button>
        <video controls autoplay>
          <source src={activeVideo.src} type="video/mp4" />
        </video>
      </div>
    </div>
  {/if}
</section>
